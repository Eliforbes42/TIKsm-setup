from "kube_pod_status_phase" 
where "phase" = 'Failed' and "gauge" = 1 and time > now() - 60s

from "docker_container_status" where "container_status" <> 'running'

from "docker_container_status" where "container_status" = 'running' 
and gauge = 1 limit 20
